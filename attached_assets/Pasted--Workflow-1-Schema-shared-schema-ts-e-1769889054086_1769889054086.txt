=== 1. Schema في shared/schema.ts ===

- enum NotificationType:
  // إشعارات سلسلة العمل
  stage_changed (تغيرت المرحلة)
  case_assigned (تم تعيين قضية)
  consultation_assigned (تم تعيين استشارة)
  sent_to_review (أُرسل للمراجعة)
  review_notes_added (أُضيفت ملاحظات المراجعة)
  returned_for_revision (أُرجع للتعديل)
  submitted_to_court (رُفع للمحكمة)
  sent_to_client (أُرسل للعميل)
  
  // إشعارات SLA
  sla_warning (تحذير اقتراب الموعد)
  sla_overdue (تأخر عن الموعد)
  sla_critical (تأخر حرج - أكثر من 24 ساعة)
  
  // إشعارات الإرجاع
  first_return (إرجاع أول)
  second_return (إرجاع ثاني)
  third_return_warning (تحذير - إرجاع ثالث)
  
  // إشعارات توازن العمل
  workload_high (حمل عمل مرتفع)
  workload_critical (حمل عمل حرج)
  reassignment_suggestion (اقتراح إعادة توزيع)
  
  // إشعارات عامة
  general_alert (تنبيه عام)
  task_reminder (تذكير بمهمة)
  deadline_warning (تحذير موعد)
  manual_notification (إشعار يدوي)

- enum NotificationPriority: low, medium, high, urgent
- enum NotificationStatus: pending, sent, read, responded, escalated, archived
- enum ResponseType: completed, in_progress, need_more_time, noted

- interface Notification:
  id, type, priority, status
  title, message
  senderId, senderName (null للإشعارات التلقائية)
  isAutomatic: boolean (تلقائي أم يدوي)
  recipientId, recipientIds[]
  
  // ربط بسلسلة العمل
  relatedType: "case" | "consultation" | null
  relatedId: string | null
  relatedStage: CaseStage | ConsultationStage | null
  workflowTriggerId: string | null (معرف الحدث الذي أطلق الإشعار)
  
  isRead, readAt
  response: { type, message, respondedAt } | null
  scheduledAt: Date | null
  escalationLevel: number
  escalatedTo: userId | null
  autoEscalateAfterHours: number
  createdAt, updatedAt

- interface NotificationRule:
  id
  name
  triggerEvent: NotificationType
  conditions: {
    stages?: (CaseStage | ConsultationStage)[]
    priorities?: CasePriority[]
    departments?: string[]
    returnCountMin?: number
    slaPercentage?: number
  }
  recipients: {
    assignedEmployee: boolean
    departmentHead: boolean
    branchManager: boolean
    reviewCommittee: boolean
    customUserIds: string[]
  }
  notificationPriority: NotificationPriority
  template: { title: string, message: string }
  isActive: boolean
  autoEscalate: boolean
  escalateAfterHours: number

- interface UserNotificationPreferences:
  userId
  enableSound: boolean
  digestMode: "instant" | "daily" | "weekly"
  mutedTypes: NotificationType[]
  quietHoursStart: string | null
  quietHoursEnd: string | null
  // تفضيلات سلسلة العمل
  notifyOnAssignment: boolean
  notifyOnStageChange: boolean
  notifyOnReviewNotes: boolean
  notifyOnReturn: boolean
  notifyOnSlaWarning: boolean

// استيراد نظام الإشعارات
import { useNotifications } from './notifications-context'

// في كل دالة تغيير مرحلة، أضف:

const updateCaseStage = (caseId, newStage, notes) => {
  // ... الكود الحالي ...
  
  // إطلاق إشعار تلقائي
  triggerWorkflowNotification({
    type: 'stage_changed',
    entityType: 'case',
    entityId: caseId,
    stage: newStage,
    additionalData: { notes, previousStage }
  })
}

const assignCaseToDepartment = (caseId, departmentId, assignedTo) => {
  // ... الكود الحالي ...
  
  triggerWorkflowNotification({
    type: 'case_assigned',
    entityType: 'case',
    entityId: caseId,
    additionalData: { assignedTo, departmentId }
  })
}

const sendCaseToReview = (caseId) => {
  // ... الكود الحالي ...
  
  triggerWorkflowNotification({
    type: 'sent_to_review',
    entityType: 'case',
    entityId: caseId
  })
}

const submitReviewNotes = (caseId, notes, action) => {
  // ... الكود الحالي ...
  
  triggerWorkflowNotification({
    type: action === 'returned' ? 'returned_for_revision' : 'review_notes_added',
    entityType: 'case',
    entityId: caseId,
    additionalData: { notes, action, returnCount }
  })
  
  // إذا كان الإرجاع للمرة الثالثة
  if (action === 'returned' && returnCount >= 3) {
    triggerWorkflowNotification({
      type: 'third_return_warning',
      entityType: 'case',
      entityId: caseId
    })
  }
}

// فحص SLA دوري
const checkSLAStatus = () => {
  cases.forEach(case => {
    const slaPercentage = calculateSLAPercentage(case)
    
    if (slaPercentage >= 80 && slaPercentage < 100) {
      triggerWorkflowNotification({
        type: 'sla_warning',
        entityType: 'case',
        entityId: case.id,
        additionalData: { slaPercentage, timeRemaining }
      })
    } else if (slaPercentage >= 100) {
      triggerWorkflowNotification({
        type: 'sla_overdue',
        entityType: 'case',
        entityId: case.id,
        additionalData: { overdueHours }
      })
    }
  })
}

// فحص توازن العمل دوري
const checkWorkloadStatus = () => {
  employees.forEach(emp => {
    const workload = getEmployeeWorkload(emp.id)
    
    if (workload.activeCases > 15) {
      triggerWorkflowNotification({
        type: 'workload_critical',
        additionalData: { employeeId: emp.id, count: workload.activeCases }
      })
    } else if (workload.activeCases > 12) {
      triggerWorkflowNotification({
        type: 'workload_high',
        additionalData: { employeeId: emp.id, count: workload.activeCases }
      })
    }
  })
}

// استدعاء الفحص الدوري في App.tsx
useEffect(() => {
  const interval = setInterval(() => {
    checkSLAStatus()
    checkWorkloadStatus()
    checkAndEscalate()
  }, 60 * 60 * 1000) // كل ساعة
  
  return () => clearInterval(interval)
}, [])
