=== 1. Schema في shared/schema.ts ===

- enum UserStatus: active, inactive, on_vacation, suspended

- enum DelegationType: full (كامل), partial (جزئي)

- interface UserVacation:
  id
  userId
  startDate
  endDate
  reason: string
  delegateTo: userId | null
  delegationType: DelegationType
  autoReassign: boolean
  status: "scheduled" | "active" | "completed" | "cancelled"
  createdAt

- interface Delegation:
  id
  fromUserId
  toUserId
  startDate
  endDate
  type: DelegationType
  permissions: string[] (الصلاحيات المفوضة إذا جزئي)
  reason: string
  isActive: boolean
  createdAt

- interface Team:
  id
  name
  description
  departmentId
  leaderId: userId
  memberIds: userId[]
  createdAt
  updatedAt

- interface UserCustomPermission:
  id
  userId
  additionalPermissions: string[]
  restrictedPermissions: string[]
  reason: string
  grantedBy: userId
  expiresAt: Date | null
  createdAt

- interface ActivityLog:
  id
  userId
  action: string
  entityType: "case" | "consultation" | "user" | "notification" | "system"
  entityId: string | null
  details: object
  ipAddress: string
  timestamp: Date

- interface UserSession:
  id
  userId
  loginAt: Date
  logoutAt: Date | null
  duration: number | null
  isActive: boolean

- في interface User أضف:
  status: UserStatus
  phone: string
  avatar: string | null
  departmentId: string
  teamId: string | null
  supervisorId: userId | null
  hireDate: Date
  lastLoginAt: Date
  currentVacation: UserVacation | null
  activeDelegations: Delegation[]
  customPermissions: UserCustomPermission | null
  
  // إحصائيات (محسوبة)
  stats: {
    activeCases: number
    activeConsultations: number
    completedThisMonth: number
    avgCompletionDays: number
    reviewAcceptanceRate: number
    returnCount: number
  }

- toggleUserStatus(id, status)
