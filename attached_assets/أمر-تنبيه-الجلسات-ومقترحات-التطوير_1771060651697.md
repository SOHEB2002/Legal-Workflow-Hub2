## أولاً: نظام التنبيه التلقائي للجلسات غير المحدّثة (مطلوب)

أحتاج نظام تنبيه تلقائي يعمل في الخلفية يراقب الجلسات التي مضى موعدها ولم تُحدَّث نتيجتها.

### 1. تثبيت مكتبة الجدولة

```bash
npm install node-cron
npm install @types/node-cron --save-dev
```

### 2. إنشاء ملف المجدوِل `server/scheduler.ts`

```typescript
import cron from "node-cron";
import { storage } from "./storage";

// المهمة تعمل كل ساعة
export function startScheduler() {
  console.log("✅ Scheduler started - checking hearings every hour");

  // فحص الجلسات غير المحدّثة كل ساعة
  cron.schedule("0 * * * *", async () => {
    console.log("🔍 Checking for unupdated hearings...");
    await checkUnupdatedHearings();
  });

  // فحص تذكيرات الجلسات القادمة (48 و 24 ساعة) كل ساعة
  cron.schedule("0 * * * *", async () => {
    await checkUpcomingHearingReminders();
  });

  // فحص مواعيد المذكرات المقتربة كل 6 ساعات
  cron.schedule("0 */6 * * *", async () => {
    await checkMemoDeadlines();
  });
}
```

### 3. دالة فحص الجلسات غير المحدّثة

```typescript
async function checkUnupdatedHearings() {
  try {
    const allHearings = await storage.getHearings();
    const now = new Date();

    for (const hearing of allHearings) {
      // فقط الجلسات بحالة "قادمة" (لم تُسجَّل نتيجتها)
      if (hearing.status !== "قادمة") continue;

      // حساب الفرق بين الآن وموعد الجلسة
      const hearingDateTime = parseHearingDateTime(hearing.hearingDate, hearing.hearingTime);
      if (!hearingDateTime) continue;

      const hoursSinceHearing = (now.getTime() - hearingDateTime.getTime()) / (1000 * 60 * 60);

      // إذا مضى 8 ساعات على موعد الجلسة ولم تُحدَّث
      if (hoursSinceHearing >= 8 && hoursSinceHearing < 9) {
        await sendUnupdatedHearingAlert(hearing);
      }

      // تصعيد: إذا مضى 24 ساعة ولم تُحدَّث
      if (hoursSinceHearing >= 24 && hoursSinceHearing < 25) {
        await sendEscalatedHearingAlert(hearing);
      }

      // تصعيد أخير: إذا مضى 48 ساعة
      if (hoursSinceHearing >= 48 && hoursSinceHearing < 49) {
        await sendFinalEscalationAlert(hearing);
      }
    }
  } catch (error) {
    console.error("Error checking unupdated hearings:", error);
  }
}
```

### 4. دالة تحويل التاريخ والوقت

```typescript
function parseHearingDateTime(dateStr: string, timeStr: string): Date | null {
  try {
    // التعامل مع التنسيقات المختلفة (ISO أو عربي)
    const date = new Date(dateStr);
    if (isNaN(date.getTime())) return null;

    if (timeStr) {
      const timeParts = timeStr.match(/(\d{1,2}):(\d{2})/);
      if (timeParts) {
        date.setHours(parseInt(timeParts[1]), parseInt(timeParts[2]));
      }
    }
    return date;
  } catch {
    return null;
  }
}
```

### 5. دوال إرسال التنبيهات

```typescript
// تنبيه بعد 8 ساعات - للمترافع ورئيس القسم والدعم الإداري
async function sendUnupdatedHearingAlert(hearing: any) {
  const caseInfo = await storage.getCase(hearing.caseId);
  if (!caseInfo) return;

  // قائمة المستلمين: المترافع + رئيس قسمه + الدعم الإداري
  const recipientIds: string[] = [];

  // 1. المحامي المترافع (المسؤول عن القضية)
  if (caseInfo.responsibleLawyerId) {
    recipientIds.push(caseInfo.responsibleLawyerId);
  }

  // 2. رئيس القسم
  if (caseInfo.responsibleLawyerId) {
    const lawyer = await storage.getUser(caseInfo.responsibleLawyerId);
    if (lawyer?.departmentId) {
      const departmentUsers = await storage.getUsersByDepartment(lawyer.departmentId);
      const deptHead = departmentUsers.find(u =>
        u.role === "department_head" || u.role === "cases_review_head"
      );
      if (deptHead) recipientIds.push(deptHead.id);
    }
  }

  // 3. الدعم الإداري (branch_manager)
  const allUsers = await storage.getUsers();
  const admins = allUsers.filter(u =>
    u.role === "branch_manager" || u.role === "admin_assistant"
  );
  admins.forEach(a => recipientIds.push(a.id));

  // إنشاء التنبيه لكل مستلم
  const uniqueRecipients = [...new Set(recipientIds)];
  for (const recipientId of uniqueRecipients) {
    await storage.createNotification({
      type: "hearing_update_overdue",
      title: `⚠️ تنبيه عاجل: جلسة لم تُحدَّث نتيجتها`,
      message: `الجلسة المتعلقة بالقضية "${caseInfo.caseTitle}" (رقم ${caseInfo.caseNumber}) بتاريخ ${hearing.hearingDate} مضى عليها أكثر من 8 ساعات ولم يتم تحديث حالتها أو تسجيل نتيجتها. يرجى تحديث حالة الجلسة فوراً.`,
      priority: "urgent",
      status: "pending",
      senderId: "system",
      senderName: "النظام التلقائي",
      recipientId: recipientId,
      relatedType: "hearing",
      relatedId: hearing.id,
      requiresResponse: true,
      autoEscalateAfterHours: 16,
    });
  }
}

// تصعيد بعد 24 ساعة - لمدير الفرع
async function sendEscalatedHearingAlert(hearing: any) {
  const caseInfo = await storage.getCase(hearing.caseId);
  if (!caseInfo) return;

  const allUsers = await storage.getUsers();
  const branchManager = allUsers.find(u => u.role === "branch_manager");

  if (branchManager) {
    await storage.createNotification({
      type: "hearing_update_overdue",
      title: `🔴 تصعيد: جلسة متأخرة التحديث 24 ساعة`,
      message: `تصعيد: الجلسة المتعلقة بالقضية "${caseInfo.caseTitle}" (رقم ${caseInfo.caseNumber}) مضى عليها 24 ساعة بدون تحديث. يرجى المتابعة مع المترافع.`,
      priority: "urgent",
      status: "pending",
      senderId: "system",
      senderName: "النظام التلقائي",
      recipientId: branchManager.id,
      relatedType: "hearing",
      relatedId: hearing.id,
      requiresResponse: true,
      escalationLevel: 1,
    });
  }
}

// تصعيد أخير بعد 48 ساعة
async function sendFinalEscalationAlert(hearing: any) {
  const caseInfo = await storage.getCase(hearing.caseId);
  if (!caseInfo) return;

  const allUsers = await storage.getUsers();
  const topManagement = allUsers.filter(u =>
    u.role === "branch_manager" || u.role === "cases_review_head"
  );

  for (const manager of topManagement) {
    await storage.createNotification({
      type: "hearing_update_overdue",
      title: `🚨 تصعيد نهائي: جلسة متأخرة 48 ساعة`,
      message: `تصعيد نهائي: الجلسة المتعلقة بالقضية "${caseInfo.caseTitle}" لم تُحدَّث منذ 48 ساعة. هذا يؤثر على سير القضية ويتطلب تدخل إداري فوري.`,
      priority: "urgent",
      status: "pending",
      senderId: "system",
      senderName: "النظام التلقائي",
      recipientId: manager.id,
      relatedType: "hearing",
      relatedId: hearing.id,
      requiresResponse: true,
      escalationLevel: 2,
    });
  }
}
```

### 6. دالة تذكيرات الجلسات القادمة

```typescript
async function checkUpcomingHearingReminders() {
  try {
    const allHearings = await storage.getHearings();
    const now = new Date();

    for (const hearing of allHearings) {
      if (hearing.status !== "قادمة") continue;

      const hearingDateTime = parseHearingDateTime(hearing.hearingDate, hearing.hearingTime);
      if (!hearingDateTime) continue;

      const hoursUntilHearing = (hearingDateTime.getTime() - now.getTime()) / (1000 * 60 * 60);

      // تذكير قبل 48 ساعة
      if (hoursUntilHearing > 47 && hoursUntilHearing <= 48 && !hearing.reminderSent24h) {
        const caseInfo = await storage.getCase(hearing.caseId);
        if (caseInfo?.responsibleLawyerId) {
          await storage.createNotification({
            type: "hearing_reminder_48h",
            title: `📅 تذكير: جلسة بعد يومين`,
            message: `تذكير: لديك جلسة في القضية "${caseInfo.caseTitle}" بتاريخ ${hearing.hearingDate} الساعة ${hearing.hearingTime}.`,
            priority: "high",
            status: "pending",
            senderId: "system",
            senderName: "النظام التلقائي",
            recipientId: caseInfo.responsibleLawyerId,
            relatedType: "hearing",
            relatedId: hearing.id,
          });
        }
      }

      // تذكير قبل 24 ساعة
      if (hoursUntilHearing > 23 && hoursUntilHearing <= 24 && !hearing.reminderSent24h) {
        const caseInfo = await storage.getCase(hearing.caseId);
        if (caseInfo?.responsibleLawyerId) {
          await storage.createNotification({
            type: "hearing_reminder_24h",
            title: `⏰ تذكير عاجل: جلسة غداً`,
            message: `تذكير عاجل: لديك جلسة غداً في القضية "${caseInfo.caseTitle}" الساعة ${hearing.hearingTime}. تأكد من جاهزية جميع المستندات.`,
            priority: "urgent",
            status: "pending",
            senderId: "system",
            senderName: "النظام التلقائي",
            recipientId: caseInfo.responsibleLawyerId,
            relatedType: "hearing",
            relatedId: hearing.id,
          });
          // تحديث علامة الإرسال
          await storage.updateHearing(hearing.id, { reminderSent24h: true });
        }
      }
    }
  } catch (error) {
    console.error("Error checking hearing reminders:", error);
  }
}
```

### 7. دالة فحص مواعيد المذكرات

```typescript
async function checkMemoDeadlines() {
  try {
    const allMemos = await storage.getMemos();
    const now = new Date();

    for (const memo of allMemos) {
      // فقط المذكرات النشطة (ليست مكتملة أو ملغية)
      if (memo.status === "مكتمل" || memo.status === "ملغي") continue;
      if (!memo.deadline) continue;

      const deadline = new Date(memo.deadline);
      const daysUntilDeadline = (deadline.getTime() - now.getTime()) / (1000 * 60 * 60 * 24);

      // تنبيه قبل 3 أيام
      if (daysUntilDeadline > 2 && daysUntilDeadline <= 3) {
        await storage.createNotification({
          type: "memo_deadline_3_days",
          title: `📝 تنبيه: موعد تسليم مذكرة بعد 3 أيام`,
          message: `المذكرة "${memo.title}" موعد تسليمها بعد 3 أيام (${memo.deadline}). يرجى الإسراع في إنجازها.`,
          priority: "high",
          status: "pending",
          senderId: "system",
          senderName: "النظام التلقائي",
          recipientId: memo.assignedTo || memo.createdBy,
          relatedType: "memo",
          relatedId: memo.id,
        });
      }

      // تنبيه قبل يوم واحد
      if (daysUntilDeadline > 0 && daysUntilDeadline <= 1) {
        await storage.createNotification({
          type: "memo_deadline_1_day",
          title: `⚠️ عاجل: موعد تسليم مذكرة غداً`,
          message: `المذكرة "${memo.title}" موعد تسليمها غداً! يرجى إنهاؤها وتسليمها فوراً.`,
          priority: "urgent",
          status: "pending",
          senderId: "system",
          senderName: "النظام التلقائي",
          recipientId: memo.assignedTo || memo.createdBy,
          relatedType: "memo",
          relatedId: memo.id,
        });
      }

      // تنبيه تأخر
      if (daysUntilDeadline < 0) {
        const daysOverdue = Math.abs(Math.floor(daysUntilDeadline));
        // تنبيه يومي للمذكرات المتأخرة (حتى 7 أيام)
        if (daysOverdue <= 7 && daysOverdue === Math.abs(Math.floor(daysUntilDeadline))) {
          await storage.createNotification({
            type: "memo_overdue",
            title: `🔴 مذكرة متأخرة ${daysOverdue} يوم`,
            message: `المذكرة "${memo.title}" تأخرت ${daysOverdue} يوم عن موعد التسليم. يرجى المعالجة فوراً.`,
            priority: "urgent",
            status: "pending",
            senderId: "system",
            senderName: "النظام التلقائي",
            recipientId: memo.assignedTo || memo.createdBy,
            relatedType: "memo",
            relatedId: memo.id,
          });
        }
      }
    }
  } catch (error) {
    console.error("Error checking memo deadlines:", error);
  }
}
```

### 10. عرض التنبيه الأحمر في الواجهة

في `client/src/components/app-sidebar.tsx` أو أي مكون يعرض الإشعارات:
- الإشعارات من نوع `hearing_update_overdue` تُعرض بخلفية حمراء وأيقونة تحذير
- تُعرض في أعلى قائمة الإشعارات (الأحدث أولاً)
- لا تختفي حتى يتم تحديث حالة الجلسة فعلاً
- أضف بادج أحمر على أيقونة الإشعارات يعرض عدد التنبيهات العاجلة

```typescript
  return await db.select().from(notifications).where(eq(notifications.relatedId, relatedId));
```
